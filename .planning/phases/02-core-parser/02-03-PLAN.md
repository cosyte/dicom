## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| dataset bytes → ByteCursor reads | Untrusted dataset bytes drive every length/group/element read. A malicious file can declare lengths that overflow buffer bounds. |
| dictionary lookup → VR inference | `Dictionary.lookup` returns frozen entries; VR inference is read-only — no mutation pathway. |
| private-creator map | The `ParseContext.creators` map is populated from parser-controlled bytes; values are bounded LO-VR strings. No code injection vector. |

```typescript
import { lookup as dictionaryLookup } from "../dictionary/index.js";
import type { DictionaryEntry, Tag, VR } from "../dictionary/types.js";
import { joinTag } from "../dataset/tag.js";
import { implicitVRForPrivateTagWithoutVR, privateTagNoCreator } from "./warnings.js";
import type { DicomParseWarning } from "./warnings.js";
import type { DicomPosition } from "./types.js";
import type { ParseContext } from "./types.js";

/** Resolve the VR for an Implicit VR LE element via the 5-case fallback per CONTEXT.md D-21.
 *
 * 1. Standard tag with single VR in dict → use it.
 * 2. Standard tag with multi-VR entry → use the first array entry.
 * 3. Standard tag matching a repeating-group family (`(50xx,xxxx)`, `(60xx,xxxx)`, etc.) → use family's first VR.
 * 4. Private tag (odd group) → fallback UN; emit DICOM_PRIVATE_TAG_NO_CREATOR (when no creator registered) AND DICOM_IMPLICIT_VR_FOR_PRIVATE_TAG_WITHOUT_VR (always — Phase 6 profile-supplied VR will resolve).
 * 5. Unknown standard tag (not in dict, not a repeating-group family) → fallback UN silently (allowed by spec).
 *
 * @internal
 */
export function resolveImplicitVR(
  tag: Tag,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
  position: DicomPosition,
): VR {
  // Case 4 first — private (odd-group) tags never consult the dictionary by tag (they're vendor-specific).
  const group = parseInt(tag.slice(0, 4), 16);
  const isPrivate = group % 2 === 1;
  if (isPrivate) {
    // Skip the (gggg,0010..00FF) block — those are Private Creator declarations themselves (LO-VR, fixed).
    const element = parseInt(tag.slice(4, 8), 16);
    if (element >= 0x0010 && element <= 0x00FF) {
      // Private Creator slot — VR is always LO per PS3.5 §7.8.
      return "LO";
    }
    const creator = resolvePrivateCreator(tag, ctx);
    if (creator === undefined) {
      emit(privateTagNoCreator(position, tag));
    }
    // Phase 6 may override based on profile dictionary; Phase 2 always falls back UN here.
    emit(implicitVRForPrivateTagWithoutVR(position, tag));
    return "UN";
  }

  // Cases 1, 2, 3, 5 — standard tag.
  const direct = dictionaryLookup(tag);
  if (direct !== undefined) {
    // Case 1 / 2: dict has a direct entry. Pick first VR (multi-VR ambiguity documented in JSDoc).
    return direct.vr[0] ?? "UN";
  }
  // Case 3: try repeating-group family match.
  const family = matchRepeatingGroup(tag);
  if (family !== undefined) {
    return family.vr[0] ?? "UN";
  }
  // Case 5: unknown standard tag — silent UN fallback.
  return "UN";
}

/** Resolve the registered Private Creator for a private element via the block-reservation rule per PS3.5 §7.8 + PITFALLS §7.1. Returns undefined for non-private tags or when no creator is registered for the element's block. */
export function resolvePrivateCreator(tag: Tag, ctx: ParseContext): string | undefined {
  const group = parseInt(tag.slice(0, 4), 16);
  if (group % 2 === 0) return undefined;
  const element = parseInt(tag.slice(4, 8), 16);
  if (element < 0x1000 || element > 0xFFFF) return undefined;
  // Element (gggg,EEFF) with 0x10 ≤ EE ≤ 0xFF is owned by Private Creator (gggg,00EE).
  const blockId = (element >> 8) & 0xFF;
  if (blockId < 0x10 || blockId > 0xFF) return undefined;
  return ctx.creators.get(group)?.get(blockId);
}

/** Register a Private Creator declaration `(gggg,00XX) VR=LO` into `ctx.creators`. Called by parsers when a creator element is parsed. */
export function registerPrivateCreator(tag: Tag, value: Buffer, ctx: ParseContext): void {
  const group = parseInt(tag.slice(0, 4), 16);
  if (group % 2 === 0) return;
  const element = parseInt(tag.slice(4, 8), 16);
  if (element < 0x0010 || element > 0x00FF) return;
  // Trim trailing space / NUL padding per PS3.5 LO conventions.
  let end = value.length;
  while (end > 0 && (value[end - 1] === 0x20 || value[end - 1] === 0x00)) end--;
  const creatorString = value.subarray(0, end).toString("ascii");
  if (creatorString.length === 0) return;
  let inner = ctx.creators.get(group);
  if (inner === undefined) {
    inner = new Map<number, string>();
    ctx.creators.set(group, inner);
  }
  inner.set(element, creatorString);
}

/** Memoized list of repeating-group family entries from the dictionary. Computed once on first call. */
let cachedFamilyEntries: readonly DictionaryEntry[] | undefined;

function getFamilyEntries(): readonly DictionaryEntry[] {
  if (cachedFamilyEntries !== undefined) return cachedFamilyEntries;
  // The dictionary's `lookup` resolves only concrete tags; we need to enumerate all entries.
  // Phase 1 frozen TAGS map is internal to dictionary/index.ts. Workaround: import the generated TAGS map directly here so we can filter by `repeatingGroup`. The TAGS map is module-level and frozen; this is a read-only access pattern.
  // eslint-disable-next-line @typescript-eslint/no-require-imports — using ESM dynamic import shape via the generated module.
  const tagsMod = require("../dictionary/generated/tags.js") as { TAGS: { [k: string]: DictionaryEntry } };
  const out: DictionaryEntry[] = [];
  for (const k of Object.keys(tagsMod.TAGS)) {
    const e = tagsMod.TAGS[k];
    if (e !== undefined && e.repeatingGroup === true) out.push(e);
  }
  cachedFamilyEntries = Object.freeze(out);
  return cachedFamilyEntries;
}

/** Match a concrete tag against any repeating-group family pattern from the dictionary. The family pattern's `tag` field carries lowercase 'x' placeholders that are wildcards. */
export function matchRepeatingGroup(tag: Tag): DictionaryEntry | undefined {
  const families = getFamilyEntries();
  for (const fam of families) {
    if (matchesFamilyPattern(fam.tag, tag)) return fam;
  }
  return undefined;
}

function matchesFamilyPattern(pattern: string, concrete: Tag): boolean {
  if (pattern.length !== 8 || concrete.length !== 8) return false;
  for (let i = 0; i < 8; i++) {
    const p = pattern[i];
    const c = concrete[i];
    if (p === undefined || c === undefined) return false;
    if (p === "x" || p === "X") continue;
    if (p.toUpperCase() !== c.toUpperCase()) return false;
  }
  return true;
}
```

import { Buffer } from "node:buffer";
import { ByteCursor } from "./byte-cursor.js";
import { Element } from "../dataset/element.js";
import { joinTag } from "../dataset/tag.js";
import {
  registerPrivateCreator,
  resolveImplicitVR,
  resolvePrivateCreator,
} from "./element-header.js";
import { groupLengthInDataset } from "./warnings.js";
import type { DicomParseWarning } from "./warnings.js";
import { DicomParseError, FATAL_CODES, buildSnippet } from "./errors.js";
import type { ParseContext } from "./types.js";
import type { Tag } from "../dictionary/types.js";

/** Implicit VR Little Endian dataset parser per CONTEXT.md D-21 + REQUIREMENTS TS-01.
 *
 * Element layout: group(2 LE) + element(2 LE) + length(4 LE) + value. VR is inferred via the 5-case fallback in resolveImplicitVR.
 *
 * @internal
 */
export function parseImplicitLE(
  buffer: Buffer,
  datasetStart: number,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
): { elements: ReadonlyMap<Tag, Element> } {
  const cursor = new ByteCursor(buffer, true, datasetStart);
  const elements = new Map<Tag, Element>();

  while (cursor.remaining() > 0) {
    const headerStart = cursor.position;
    let group: number;
    let element: number;
    let length: number;
    try {
      group = cursor.readUInt16();
      element = cursor.readUInt16();
      length = cursor.readUInt32();
    } catch (err) {
      if (err instanceof RangeError) {
        throw new DicomParseError(
          FATAL_CODES.INVALID_FILE_META,
          "Truncated dataset (header read past buffer end)",
          headerStart,
          buildSnippet(buffer, headerStart),
        );
      }
      throw err;
    }
    const tag = joinTag(group, element);

    // FFFE markers under Implicit VR LE — Item / ItemDelim / SeqDelim. Encoding-context stack determines semantics; SQ-aware logic lives in parser/sequence.ts (plan 02-04). For root-level Implicit-LE parsing, FFFE markers should not appear unless we're descending an SQ — which means we'd be inside parseSequence already. Bail out: throw INVALID_FILE_META if encountered at root unexpectedly.
    if (group === 0xFFFE) {
      // Root-level FFFE marker is malformed per PS3.5 §7.5 — it must be inside an SQ. parseSequence (plan 02-04) handles in-SQ FFFE markers.
      throw new DicomParseError(
        FATAL_CODES.INVALID_FILE_META,
        `Unexpected FFFE marker (${tag}) at dataset root.`,
        headerStart,
        buildSnippet(buffer, headerStart),
      );
    }

    const position = { byteOffset: headerStart };
    const vr = resolveImplicitVR(tag, ctx, emit, position);

    // length === 0xFFFFFFFF — under Implicit VR LE, only valid for SQ (delegate to parseSequence).
    if (length === 0xFFFFFFFF) {
      if (vr === "SQ") {
        // Delegate to parseSequence — plan 02-04 owns parser/sequence.ts.
        const seq = parseSequence(buffer, cursor.position, ctx, emit, /* isImplicitVR */ true);
        // Construct the SQ Element with raw bytes spanning the cursor before-and-after.
        const valueStart = cursor.position;
        cursor.position = seq.endOffset;
        const rawBytes = ctx.copyValues
          ? Buffer.from(buffer.subarray(headerStart, cursor.position))
          : buffer.subarray(headerStart, cursor.position);
        elements.set(tag, new Element({
          tag, vr, vm: seq.items.length, length: 0xFFFFFFFF, rawBytes, byteOffset: headerStart,
          ...(resolvePrivateCreator(tag, ctx) !== undefined ? { privateCreator: resolvePrivateCreator(tag, ctx) } : {}),
        }));
        continue;
      }
      // Implicit VR LE has no on-wire VR; UN cannot legitimately appear with undefined length.
      throw new DicomParseError(
        FATAL_CODES.INVALID_FILE_META,
        `Undefined length on non-SQ element ${tag} (vr=${vr}) under Implicit VR LE.`,
        headerStart,
        buildSnippet(buffer, headerStart),
      );
    }

    // Bounds check (T-02-03-02).
    if (cursor.position + length > buffer.length) {
      throw new DicomParseError(
        FATAL_CODES.INVALID_FILE_META,
        `Element ${tag} length=${length} exceeds remaining buffer (${buffer.length - cursor.position} bytes).`,
        headerStart,
        buildSnippet(buffer, headerStart),
      );
    }

    const valueSlice = ctx.copyValues
      ? Buffer.from(buffer.subarray(cursor.position, cursor.position + length))
      : buffer.subarray(cursor.position, cursor.position + length);
    cursor.position += length;

    // Group-length in non-FM groups (TOL-10).
    if (element === 0x0000 && group !== 0x0002) {
      emit(groupLengthInDataset(position, tag));
    }

    // Private Creator slot (gggg,0010..00FF) → register.
    if (group % 2 === 1 && element >= 0x0010 && element <= 0x00FF) {
      registerPrivateCreator(tag, valueSlice, ctx);
    }

    const privateCreator = resolvePrivateCreator(tag, ctx);
    elements.set(tag, new Element({
      tag,
      vr,
      vm: 1, // Phase 2 placeholder — Phase 3 owns VM derivation from VR + value layout.
      length,
      rawBytes: valueSlice,
      byteOffset: headerStart,
      ...(privateCreator !== undefined ? { privateCreator } : {}),
    }));
  }

  return { elements };
}
