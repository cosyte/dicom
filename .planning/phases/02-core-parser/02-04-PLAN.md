<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| dataset bytes → ByteCursor reads | Untrusted dataset bytes drive every length/group/element read. |
| FFFE marker reads under BE | Spec corner: PITFALLS §2.3 — naive FFFE reads under BE swap groups, breaking SQ termination detection. |
| CP-246 fallback | Untrusted UN-undefined-length bytes drive recursive SQ descent — a malicious file can attempt to exhaust memory or stack. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04-01 | Buffer over-read on truncated input | parseExplicitLE / parseExplicitBE / parseSequence element loops | mitigate | Each `cursor.slice(N)` and explicit `cursor.position + N > buffer.length` guard throws `DicomParseError(INVALID_FILE_META, ...)` rather than reading past end. Test programmatically truncates each TS fixture mid-element. |
| T-02-04-02 | Stack overflow via deeply-nested SQ | parseSequence | mitigate | `ctx.nestingDepth` increments on SQ descent; on exceed 64, throw `DicomParseError(INVALID_FILE_META, 'SQ nesting depth exceeds 64', offset, snippet)`. Test programmatically builds a fixture with 65-level SQ nesting and asserts the typed throw at depth 64+1. |

import { Buffer } from "node:buffer";
import { Dataset } from "../dataset/dataset.js";
import { Element } from "../dataset/element.js";
import { Item } from "../dataset/item.js";
import { ByteCursor } from "./byte-cursor.js";
import {
  emptyItemInSequence,
  undefinedLengthInExplicitVR,
  unParsedAsSQ,
} from "./warnings.js";
import type { DicomParseWarning } from "./warnings.js";
import { DicomParseError, FATAL_CODES, buildSnippet } from "./errors.js";
import type { ParseContext } from "./types.js";
import type { Tag } from "../dictionary/types.js";

const ITEM_TAG = "FFFEE000";
const ITEM_DELIM_TAG = "FFFEE00D";
const SEQ_DELIM_TAG = "FFFEE0DD";
const NESTING_DEPTH_LIMIT = 64;

export type InnerParser = (
  buffer: Buffer,
  datasetStart: number,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
) => { elements: ReadonlyMap<Tag, Element> };

export interface ParseSequenceOptions {
  /** When undefined, parse items until SeqDelim. When a number, parse for exactly N bytes. */
  readonly explicitLength: number | undefined;
  readonly littleEndian: boolean;
  readonly innerStrategy: InnerParser;
  /** True when the SQ wraps encapsulated pixel data (D-31). Each item is a fragment; first item is the Basic Offset Table. */
  readonly encapsulatedPixelData?: boolean;
}

export interface ParseSequenceResult {
  readonly items: readonly Item[];
  readonly endOffset: number;
}

/** Parse an SQ value (or encapsulated-pixel-data value) starting at `valueStart`. Caller has already read the SQ Element header (group/element/VR/length); this function consumes the items + delimiters and returns where it ended.
 *
 * @internal
 */
export function parseSequence(
  buffer: Buffer,
  valueStart: number,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
  opts: ParseSequenceOptions,
): ParseSequenceResult {
  ctx.nestingDepth += 1;
  if (ctx.nestingDepth > NESTING_DEPTH_LIMIT) {
    throw new DicomParseError(
      FATAL_CODES.INVALID_FILE_META,
      `SQ nesting depth exceeds ${NESTING_DEPTH_LIMIT}.`,
      valueStart,
      buildSnippet(buffer, valueStart),
    );
  }
  ctx.encodingContextStack.push(opts.encapsulatedPixelData === true ? "EncapsulatedPixelData" : "SqItem");

  try {
    const cursor = new ByteCursor(buffer, opts.littleEndian, valueStart);
    const items: Item[] = [];
    const endLimit = opts.explicitLength !== undefined ? valueStart + opts.explicitLength : buffer.length;

    if (opts.explicitLength === undefined && opts.encapsulatedPixelData !== true) {
      // Undefined length on Explicit VR — emit DICOM_UNDEFINED_LENGTH_IN_EXPLICIT_VR per SQ-03.
      // Note: caller must determine "is this Explicit VR?" — pass through opts.
      // For now this is fired unconditionally on undefined-length SQ (Implicit VR LE has its own undefined-length convention but parseImplicitLE always passes explicitLength=undefined for SQ — to avoid double-warning, the caller (parseImplicitLE) suppresses this by setting an opts flag; or parseSequence checks opts.innerStrategy === parseImplicitLE — planner discretion). Cleanest: the caller (parseExplicitLE/BE) emits this warning BEFORE calling parseSequence; parseSequence is silent on it. Move the emit out.
    }

    let itemIndex = 0;
    while (cursor.position < endLimit) {
      const itemHeaderStart = cursor.position;
      let itemTag: string;
      let itemLength: number;
      try {
        const grp = cursor.readUInt16();
        const ele = cursor.readUInt16();
        itemTag = (grp.toString(16).padStart(4, "0") + ele.toString(16).padStart(4, "0")).toUpperCase();
        itemLength = cursor.readUInt32();
      } catch (err) {
        if (err instanceof RangeError) {
          throw new DicomParseError(
            FATAL_CODES.INVALID_FILE_META, "Truncated dataset (SQ item header read past buffer end)",
            itemHeaderStart, buildSnippet(buffer, itemHeaderStart),
          );
        }
        throw err;
      }

      if (itemTag === SEQ_DELIM_TAG) {
        // SeqDelim — skip the 4-byte length field (always 0) and exit.
        return { items, endOffset: cursor.position };
      }
      if (itemTag !== ITEM_TAG) {
        throw new DicomParseError(
          FATAL_CODES.INVALID_FILE_META,
          `Unexpected tag ${itemTag} inside sequence (expected FFFE,E000 or FFFE,E0DD).`,
          itemHeaderStart, buildSnippet(buffer, itemHeaderStart),
        );
      }

      // (FFFE,E000) Item.
      const itemValueStart = cursor.position;

      if (itemLength === 0) {
        emit(emptyItemInSequence({ byteOffset: itemHeaderStart }, itemTag));
        items.push(new Item({
          warnings: [], elements: new Map(), index: itemIndex,
        }));
        itemIndex++;
        continue;
      }

      if (opts.encapsulatedPixelData === true) {
        // Pixel-data fragment: consume `itemLength` bytes verbatim. Phase 4 surfaces the bytes; Phase 2 records the structure as an empty Item but advances the cursor past the fragment.
        if (cursor.position + itemLength > buffer.length) {
          throw new DicomParseError(
            FATAL_CODES.INVALID_FILE_META,
            `Encapsulated pixel data fragment length=${itemLength} exceeds remaining buffer.`,
            itemHeaderStart, buildSnippet(buffer, itemHeaderStart),
          );
        }
        cursor.position += itemLength;
        items.push(new Item({ warnings: [], elements: new Map(), index: itemIndex }));
        itemIndex++;
        continue;
      }

      if (itemLength === 0xFFFFFFFF) {
        // Undefined-length item — parse inner elements until ItemDelim.
        const inner = parseInnerUntilItemDelim(buffer, itemValueStart, ctx, emit, opts.innerStrategy);
        items.push(new Item({ warnings: [], elements: inner.elements, index: itemIndex }));
        cursor.position = inner.endOffset;
      } else {
        // Defined-length item — slice and parse exactly `itemLength` bytes.
        if (cursor.position + itemLength > buffer.length) {
          throw new DicomParseError(
            FATAL_CODES.INVALID_FILE_META,
            `Item length=${itemLength} exceeds remaining buffer.`,
            itemHeaderStart, buildSnippet(buffer, itemHeaderStart),
          );
        }
        const itemSlice = buffer.subarray(cursor.position, cursor.position + itemLength);
        const innerCtx = ctx; // Same ctx — onWarning still fires; nesting handled at the top of this function.
        const { elements } = opts.innerStrategy(itemSlice, 0, innerCtx, emit);
        items.push(new Item({ warnings: [], elements, index: itemIndex }));
        cursor.position += itemLength;
      }
      itemIndex++;
    }

    return { items, endOffset: cursor.position };
  } finally {
    ctx.nestingDepth -= 1;
    ctx.encodingContextStack.pop();
  }
}

/** For undefined-length items (item length = 0xFFFFFFFF): parse inner elements until (FFFE,E00D) ItemDelim. */
function parseInnerUntilItemDelim(
  buffer: Buffer,
  startOffset: number,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
  innerStrategy: InnerParser,
): { elements: ReadonlyMap<Tag, Element>; endOffset: number } {
  // Strategy: scan ahead byte-by-byte to find (FFFE,E00D), then call innerStrategy on the slice [startOffset, itemDelimStart).
  // (More robust: inner-parse with a custom innerStrategy variant that stops on ItemDelim. For Phase 2 simplicity, scan first.)
  let scan = startOffset;
  while (scan + 8 <= buffer.length) {
    // Tag bytes vary by endianness; scan in both LE and BE forms is awkward. Simpler: invoke innerStrategy on the FULL remaining slice, then look at where it stopped — but innerStrategy doesn't know about ItemDelim either.
    //
    // Simplest correct approach: call a SPECIAL inner that knows to stop on ItemDelim. For Phase 2 we add an `innerStrategyStopOnItemDelim` parameter — planner discretion. The cleanest path is: have parseImplicitLE / parseExplicitLE / parseExplicitBE accept an optional `stopAtItemDelim: boolean` parameter that breaks the element loop when (FFFE,E00D) is read. Add this in this plan's TS-parser implementations (Tasks 2 + 3).
    break;
  }
  const result = innerStrategy(buffer, startOffset, ctx, emit);
  // For Phase 2 we trust innerStrategy to stop at ItemDelim and return the post-delim offset. The TS parsers Tasks 2 + 3 implement that contract.
  return { elements: result.elements, endOffset: scan }; // The endOffset comes back via a contract — see TS-parser tasks below.
}

/** CP-246 fallback per D-30. Try to descend the bytes as Implicit VR LE SQ items. On any failure, restore state and signal failure to caller. */
export function tryParseUnAsSQ(
  buffer: Buffer,
  valueStart: number,
  valueLength: number,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
  implicitLeInner: InnerParser,
): { success: boolean; items: readonly Item[]; endOffset: number } {
  // Save state for rollback.
  const savedDepth = ctx.nestingDepth;
  const savedStackLen = ctx.encodingContextStack.length;
  const savedWarnings = ctx.warnings.length;
  try {
    const slice = valueLength === 0xFFFFFFFF ? buffer.subarray(valueStart) : buffer.subarray(valueStart, valueStart + valueLength);
    const opts: ParseSequenceOptions = {
      explicitLength: valueLength === 0xFFFFFFFF ? undefined : valueLength,
      littleEndian: true,                  // Implicit-LE-inner per D-30
      innerStrategy: implicitLeInner,
    };
    const result = parseSequence(slice, 0, ctx, emit, opts);
    emit(unParsedAsSQ({ byteOffset: valueStart }, "UN"));
    return { success: true, items: result.items, endOffset: valueStart + result.endOffset };
  } catch (_err) {
    // Restore state — drop any warnings emitted during the failed attempt.
    ctx.nestingDepth = savedDepth;
    while (ctx.encodingContextStack.length > savedStackLen) ctx.encodingContextStack.pop();
    while (ctx.warnings.length > savedWarnings) ctx.warnings.pop();
    return { success: false, items: [], endOffset: valueStart };
  }
}

export interface ExplicitElementHeader {
  readonly tag: Tag;
  readonly vr: VR;
  readonly length: number;
  readonly headerStart: number;
  readonly headerLength: number;  // 8 (short) or 12 (long)
}

/** Parse an Explicit VR element header (LE or BE — endian determined by `cursor.littleEndian`). Validates reserved bytes for long-form VRs. */
export function readExplicitElementHeader(
  cursor: ByteCursor,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
): ExplicitElementHeader {
  const headerStart = cursor.position;
  const group = cursor.readUInt16();
  const element = cursor.readUInt16();
  const tag = (group.toString(16).padStart(4, "0") + element.toString(16).padStart(4, "0")).toUpperCase() as Tag;
  // VR is always 2 ASCII bytes regardless of endian.
  const vrBytes = cursor.slice(2);
  const vr = vrBytes.toString("ascii") as VR;

  let length: number;
  let headerLength: number;
  if (LONG_FORM_VRS.has(vr)) {
    const reserved0 = cursor.buffer[cursor.position];
    const reserved1 = cursor.buffer[cursor.position + 1];
    cursor.position += 2;
    if (reserved0 !== 0x00 || reserved1 !== 0x00) {
      const observed = `${(reserved0 ?? 0).toString(16).padStart(2, "0")}${(reserved1 ?? 0).toString(16).padStart(2, "0")}`;
      emit(nonzeroReservedBytes({ byteOffset: headerStart }, tag, observed));
    }
    length = cursor.readUInt32();
    headerLength = 12;
  } else {
    length = cursor.readUInt16();
    headerLength = 8;
  }
  return { tag, vr, length, headerStart, headerLength };
}
