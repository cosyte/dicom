## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| consumer call → parser type surface | All public types (`ParseOptions`, `DicomParseWarning`, `DicomParseError`) cross from untrusted consumer code into the library. Type-surface integrity is the primary mitigation; bad shapes are filtered by TS at compile time and at-runtime by ignored fields. |
| consumer-supplied `onWarning` callback | Callback can throw, return a Promise, or block. The `emit` chokepoint MUST silently swallow handler exceptions per D-03 + sibling pattern, never let a noisy handler corrupt parser state. |
| internal modules → public exports | Phase 2 must not leak internal helpers (`ParseContext`, `RawElement`, `ByteCursor` impl details) into `src/index.ts`. |

```typescript
export const FATAL_CODES = {
  NOT_DICOM_PART_10: "NOT_DICOM_PART_10",
  INVALID_FILE_META: "INVALID_FILE_META",
  UNSUPPORTED_TRANSFER_SYNTAX: "UNSUPPORTED_TRANSFER_SYNTAX",
  EMPTY_INPUT: "EMPTY_INPUT",
} as const;

export type FatalCode = (typeof FATAL_CODES)[keyof typeof FATAL_CODES];

export class DicomParseError extends Error {
  public readonly code: FatalCode;
  public readonly byteOffset: number;
  public readonly snippet: string;
  public readonly contextPath: readonly string[] | undefined;

  public constructor(
    code: FatalCode,
    message: string,
    byteOffset: number,
    snippet: string,
    contextPath?: readonly string[],
  ) {
    const formatted = `[${code}] ${message} (offset=${byteOffset})`
      + (contextPath !== undefined && contextPath.length > 0
          ? ` … in ${contextPath.join("/")}`
          : "");
    super(formatted);
    this.name = "DicomParseError";
    this.code = code;
    this.byteOffset = byteOffset;
    this.snippet = snippet;
    this.contextPath = contextPath;
  }
}

/** Build a 16-byte hex snippet (space-separated, lowercase) from `buffer` at `offset`. Returns "" if offset is out of range. */
export function buildSnippet(buffer: Buffer, offset: number): string {
  if (offset < 0 || offset >= buffer.length) return "";
  const end = Math.min(offset + 16, buffer.length);
  const slice = buffer.subarray(offset, end);
  return Array.from(slice).map((b) => b.toString(16).padStart(2, "0")).join(" ");
}
```

```typescript
import type { Buffer } from "node:buffer";

/** Endian-aware Buffer cursor — single primitive shared by all 4 TS parsers (D-05). Uses Node Buffer.read* methods (NOT DataView, per CONTEXT specifics §) for performance and project idiom parity. */
export class ByteCursor {
  public readonly buffer: Buffer;
  public readonly littleEndian: boolean;
  public position: number;

  public constructor(buffer: Buffer, littleEndian: boolean, position = 0) {
    this.buffer = buffer;
    this.littleEndian = littleEndian;
    this.position = position;
  }

  public readUInt16(): number {
    if (this.position + 2 > this.buffer.length) throw new RangeError("ByteCursor: read past end of buffer");
    const v = this.littleEndian ? this.buffer.readUInt16LE(this.position) : this.buffer.readUInt16BE(this.position);
    this.position += 2;
    return v;
  }

  public readUInt32(): number {
    if (this.position + 4 > this.buffer.length) throw new RangeError("ByteCursor: read past end of buffer");
    const v = this.littleEndian ? this.buffer.readUInt32LE(this.position) : this.buffer.readUInt32BE(this.position);
    this.position += 4;
    return v;
  }

  public readUInt16At(offset: number): number {
    if (offset < 0 || offset + 2 > this.buffer.length) throw new RangeError("ByteCursor: read past end of buffer");
    return this.littleEndian ? this.buffer.readUInt16LE(offset) : this.buffer.readUInt16BE(offset);
  }

  public readUInt32At(offset: number): number {
    if (offset < 0 || offset + 4 > this.buffer.length) throw new RangeError("ByteCursor: read past end of buffer");
    return this.littleEndian ? this.buffer.readUInt32LE(offset) : this.buffer.readUInt32BE(offset);
  }

  public slice(length: number): Buffer {
    if (length < 0 || this.position + length > this.buffer.length) throw new RangeError("ByteCursor: read past end of buffer");
    const out = this.buffer.subarray(this.position, this.position + length);
    this.position += length;
    return out;
  }

  public remaining(): number {
    return this.buffer.length - this.position;
  }
}
```

```typescript
import type { DicomParseWarning } from "./warnings.js";
import type { ParseContext } from "./types.js";
import { DicomParseError, buildSnippet, type FatalCode } from "./errors.js";

/** Single chokepoint for Tier-2 warning emission per CONTEXT.md D-11 + D-35.
 *
 * Lenient (`ctx.strict === false`): push warning into `ctx.warnings` FIRST,
 * then invoke `ctx.onWarning` (if defined) inside its own try/catch — silent
 * swallow on handler throw, mirrors `@cosyte/hl7` sibling lines 158–179.
 *
 * Strict (`ctx.strict === true`): throw `DicomParseError` carrying the warning
 * code. The `code` field is typed `FatalCode` at compile time but at runtime
 * carries the `WarningCode` literal (see `as unknown as` cast — Plan 06
 * decision (b) in HL7 sibling). Consumers narrow on `err.code` after catch.
 *
 * @internal
 */
export function makeEmitter(ctx: ParseContext): (w: DicomParseWarning) => void {
  return (w) => {
    if (ctx.strict) {
      throw new DicomParseError(
        w.code as unknown as FatalCode,
        w.message,
        w.position.byteOffset,
        buildSnippet(ctx.buffer, w.position.byteOffset),
        w.position.contextPath,
      );
    }
    ctx.warnings.push(w);
    if (ctx.onWarning !== undefined) {
      try {
        ctx.onWarning(w);
      } catch {
        // D-03 silent swallow — a noisy handler must not corrupt parser state (T-02-01-02).
      }
    }
  };
}
```
