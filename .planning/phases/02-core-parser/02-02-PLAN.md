| Boundary | Description |
|----------|-------------|

**Step 1 — `src/parser/part10-header.ts`.** Skeleton:

```typescript
import type { Buffer } from "node:buffer";
import { DicomParseError, FATAL_CODES, buildSnippet } from "./errors.js";
import { missingPreamble } from "./warnings.js";
import type { DicomParseWarning } from "./warnings.js";
import type { ParseContext } from "./types.js";

const DICM_MAGIC = Buffer.from("DICM", "ascii");

export interface Part10HeaderResult {
  readonly datasetStart: number;
  readonly hadPreamble: boolean;
}

/** Detect the Part 10 preamble + DICM magic. Per CONTEXT.md D-14 + D-15.
 *
 * Behavior:
 * - DICM at offset 128 → strip 128+4 bytes; return { datasetStart: 132, hadPreamble: true }.
 * - stripPreamble === 'require' AND no DICM → throw NOT_DICOM_PART_10.
 * - stripPreamble === 'tolerate' AND no DICM:
 *     - If buffer has a plausible (0002,0000) UL element at offset 0 → emit DICOM_MISSING_PREAMBLE; return { datasetStart: 0, hadPreamble: false }.
 *     - Else → throw NOT_DICOM_PART_10.
 *
 * @internal
 */
export function parsePart10Header(
  buffer: Buffer,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
): Part10HeaderResult {
  if (buffer.length >= 132 && buffer.subarray(128, 132).equals(DICM_MAGIC)) {
    return { datasetStart: 132, hadPreamble: true };
  }

  if (ctx.stripPreamble === "require") {
    throw new DicomParseError(
      FATAL_CODES.NOT_DICOM_PART_10,
      "Missing DICM magic at offset 128 (stripPreamble='require').",
      0,
      buildSnippet(buffer, 0),
    );
  }

  // stripPreamble === 'tolerate' — try detecting (0002,0000) UL element at offset 0.
  // Header layout in Explicit VR LE: group(2) + element(2) + VR(2,'UL') + length(2,4) + value(4) = 12 bytes.
  if (looksLikeFileMetaGroupLengthAtOffsetZero(buffer)) {
    emit(missingPreamble({ byteOffset: 0 }));
    return { datasetStart: 0, hadPreamble: false };
  }

  throw new DicomParseError(
    FATAL_CODES.NOT_DICOM_PART_10,
    "Input is not a DICOM Part 10 file (no DICM magic at offset 128 and no recognizable File Meta group at offset 0).",
    0,
    buildSnippet(buffer, 0),
  );
}

/** Heuristic per CONTEXT.md D-15: detect (0002,0000) FileMetaGroupLength element at offset 0 in Explicit VR LE form. */
function looksLikeFileMetaGroupLengthAtOffsetZero(buffer: Buffer): boolean {
  if (buffer.length < 12) return false;
  // group = 0x0002 (LE: bytes 02 00); element = 0x0000 (LE: bytes 00 00); VR = 'UL' (ascii bytes 0x55 0x4C); length = 0x0004 (LE: bytes 04 00).
  return (
    buffer.readUInt16LE(0) === 0x0002 &&
    buffer.readUInt16LE(2) === 0x0000 &&
    buffer.readUInt8(4) === 0x55 && // 'U'
    buffer.readUInt8(5) === 0x4c && // 'L'
    buffer.readUInt16LE(6) === 0x0004
  );
}
```

```typescript
import { Buffer } from "node:buffer";
import type { Tag, VR } from "../../src/dictionary/types.js";

export interface BuildDicomElement {
  readonly tag: Tag;          // 8-char uppercase hex
  readonly vr: VR;
  readonly value: Buffer;     // raw value bytes (caller pads to even length when required by VR rules)
}

export interface BuildDicomOptions {
  readonly transferSyntax: string;
  readonly elements: readonly BuildDicomElement[];
  readonly skipPreamble?: boolean;
  readonly fileMetaGroupLength?: number | "omit" | "wrong";
  /** When true, omits the (0002,0010) Transfer Syntax UID element from File Meta. */
  readonly skipTransferSyntaxUID?: boolean;
  /** Optional File Meta extras (FM-02 fields beyond TS UID). */
  readonly mediaStorageSOPClassUID?: string;
  readonly mediaStorageSOPInstanceUID?: string;
  readonly implementationClassUID?: string;
  readonly implementationVersionName?: string;
}

export function buildDicom(opts: BuildDicomOptions): Buffer {
  const parts: Buffer[] = [];

  // 1. Preamble + DICM magic (unless skipped).
  if (opts.skipPreamble !== true) {
    parts.push(Buffer.alloc(128, 0x00));
    parts.push(Buffer.from("DICM", "ascii"));
  }

  // 2. Build File Meta body (Explicit VR LE).
  const fileMetaElements: Buffer[] = [];
  if (opts.skipTransferSyntaxUID !== true) {
    fileMetaElements.push(buildExplicitLeElement("00020010", "UI", padUI(opts.transferSyntax)));
  }
  if (opts.mediaStorageSOPClassUID !== undefined) {
    fileMetaElements.push(buildExplicitLeElement("00020002", "UI", padUI(opts.mediaStorageSOPClassUID)));
  }
  if (opts.mediaStorageSOPInstanceUID !== undefined) {
    fileMetaElements.push(buildExplicitLeElement("00020003", "UI", padUI(opts.mediaStorageSOPInstanceUID)));
  }
  if (opts.implementationClassUID !== undefined) {
    fileMetaElements.push(buildExplicitLeElement("00020012", "UI", padUI(opts.implementationClassUID)));
  }
  if (opts.implementationVersionName !== undefined) {
    fileMetaElements.push(buildExplicitLeElement("00020013", "SH", padText(opts.implementationVersionName)));
  }
  const fileMetaBody = Buffer.concat(fileMetaElements);

  // 3. Optional (0002,0000) FileMetaInformationGroupLength.
  if (opts.fileMetaGroupLength !== "omit") {
    let declared: number;
    if (opts.fileMetaGroupLength === "wrong") declared = fileMetaBody.length + 99; // intentionally wrong
    else if (typeof opts.fileMetaGroupLength === "number") declared = opts.fileMetaGroupLength;
    else declared = fileMetaBody.length;
    const lengthValue = Buffer.alloc(4);
    lengthValue.writeUInt32LE(declared, 0);
    parts.push(buildExplicitLeElement("00020000", "UL", lengthValue));
  }
  parts.push(fileMetaBody);

  // 4. Dataset elements per the requested transfer syntax.
  for (const el of opts.elements) {
    parts.push(encodeElement(el, opts.transferSyntax));
  }

  return Buffer.concat(parts);
}

// ---------- helpers ----------

function buildExplicitLeElement(tag: Tag, vr: VR, value: Buffer): Buffer {
  const { group, element } = splitTag(tag);
  const isLongForm = ["OB","OW","OF","OD","OL","SQ","UT","UN","UC","UR"].includes(vr);
  const groupBuf = Buffer.alloc(2); groupBuf.writeUInt16LE(group, 0);
  const elementBuf = Buffer.alloc(2); elementBuf.writeUInt16LE(element, 0);
  const vrBuf = Buffer.from(vr, "ascii");
  if (isLongForm) {
    // 2-byte VR + 2 reserved (0x00 0x00) + 4-byte length
    const reserved = Buffer.from([0x00, 0x00]);
    const lengthBuf = Buffer.alloc(4); lengthBuf.writeUInt32LE(value.length, 0);
    return Buffer.concat([groupBuf, elementBuf, vrBuf, reserved, lengthBuf, value]);
  }
  // 2-byte VR + 2-byte length
  const lengthBuf = Buffer.alloc(2); lengthBuf.writeUInt16LE(value.length, 0);
  return Buffer.concat([groupBuf, elementBuf, vrBuf, lengthBuf, value]);
}

function buildImplicitLeElement(tag: Tag, value: Buffer): Buffer {
  const { group, element } = splitTag(tag);
  const groupBuf = Buffer.alloc(2); groupBuf.writeUInt16LE(group, 0);
  const elementBuf = Buffer.alloc(2); elementBuf.writeUInt16LE(element, 0);
  const lengthBuf = Buffer.alloc(4); lengthBuf.writeUInt32LE(value.length, 0);
  return Buffer.concat([groupBuf, elementBuf, lengthBuf, value]);
}

function encodeElement(el: BuildDicomElement, ts: string): Buffer {
  if (ts === "1.2.840.10008.1.2") return buildImplicitLeElement(el.tag, el.value);
  if (ts === "1.2.840.10008.1.2.1") return buildExplicitLeElement(el.tag, el.vr, el.value);
  // Explicit VR BE encoder lands in plan 02-04; Deflated LE in plan 02-05. Throw clearly so tests fail fast until those land.
  throw new Error(`buildDicom: encoder for transferSyntax="${ts}" not implemented yet (added in plan 02-04 / 02-05)`);
}

function splitTag(tag: Tag): { group: number; element: number } {
  return { group: parseInt(tag.slice(0, 4), 16), element: parseInt(tag.slice(4, 8), 16) };
}

function padUI(s: string): Buffer {
  // UI must be even-length; pad with NULL (0x00) per PS3.5.
  const buf = Buffer.from(s, "ascii");
  return buf.length % 2 === 0 ? buf : Buffer.concat([buf, Buffer.from([0x00])]);
}

function padText(s: string): Buffer {
  // Short-string VRs (SH, etc.) pad with space (0x20) per PS3.5.
  const buf = Buffer.from(s, "ascii");
  return buf.length % 2 === 0 ? buf : Buffer.concat([buf, Buffer.from([0x20])]);
}
```

import { Buffer } from "node:buffer";
import { ByteCursor } from "./byte-cursor.js";
import { LONG_FORM_VRS } from "./element-header.js";
import { DicomParseError, FATAL_CODES, buildSnippet } from "./errors.js";
import { fileMetaGroupLengthMismatch, fileMetaGroupLengthMissing, nonzeroReservedBytes } from "./warnings.js";
import type { DicomParseWarning } from "./warnings.js";
import type { ParseContext } from "./types.js";
import type { FileMeta } from "../dataset/file-meta.js";
import type { Tag, VR } from "../dictionary/types.js";

interface FmRawElement {
  readonly tag: Tag;
  readonly vr: VR;
  readonly value: Buffer;
  readonly headerStart: number;
  readonly headerLength: number;  // header bytes consumed (8 short-form / 12 long-form)
}

/** Parse the File Meta Information group. ALWAYS Explicit VR LE per FM-01 + CONTEXT.md D-17.
 *
 * @internal
 */
export function parseFileMeta(
  buffer: Buffer,
  datasetStart: number,
  ctx: ParseContext,
  emit: (w: DicomParseWarning) => void,
): { fileMeta: FileMeta; fileMetaEnd: number } {
  const cursor = new ByteCursor(buffer, true, datasetStart);
  const fmStart = cursor.position;

  // Step 1: Read (0002,0000) FileMetaInformationGroupLength if present.
  let declaredFmLength: number | undefined;
  let firstElement: FmRawElement | undefined;
  try {
    firstElement = readExplicitLeElement(cursor);
  } catch (err) {
    if (err instanceof RangeError) {
      throw new DicomParseError(FATAL_CODES.INVALID_FILE_META, "File Meta is truncated.", fmStart, buildSnippet(buffer, fmStart));
    }
    throw err;
  }

  let consumedAfterGroupLength = 0;
  let fmElements: FmRawElement[] = [];

  if (firstElement.tag === "00020000" && firstElement.vr === "UL" && firstElement.value.length === 4) {
    declaredFmLength = firstElement.value.readUInt32LE(0);
    // Continue reading until we have exactly declaredFmLength bytes, or until we hit a non-(0002,xxxx) group.
  } else {
    // (0002,0000) absent — emit warning and treat first element as the start of File Meta body.
    emit(fileMetaGroupLengthMissing({ byteOffset: fmStart, fileMeta: true }));
    fmElements.push(firstElement);
    consumedAfterGroupLength = firstElement.headerLength + firstElement.value.length;
  }

  // Step 2: Read remaining File Meta elements.
  while (cursor.remaining() > 0) {
    const peekGroup = cursor.readUInt16At(cursor.position);
    if (peekGroup !== 0x0002) break; // first non-(0002,xxxx) group — File Meta ends here.
    const next = readExplicitLeElement(cursor);
    fmElements.push(next);
    consumedAfterGroupLength += next.headerLength + next.value.length;
    if (declaredFmLength !== undefined && consumedAfterGroupLength >= declaredFmLength) break;
  }

  // Step 3: Validate declared vs actual (D-18).
  if (declaredFmLength !== undefined && consumedAfterGroupLength !== declaredFmLength) {
    emit(fileMetaGroupLengthMismatch({ byteOffset: fmStart, fileMeta: true }, declaredFmLength, consumedAfterGroupLength));
    // Trust actual — keep parsing forward through any remaining (0002,xxxx) elements until first non-FM group.
    while (cursor.remaining() > 0) {
      const peekGroup = cursor.readUInt16At(cursor.position);
      if (peekGroup !== 0x0002) break;
      const next = readExplicitLeElement(cursor);
      fmElements.push(next);
    }
  }

  // Step 4: Project File Meta elements into the FileMeta interface (D-19 — only TS UID is required).
  const tsElement = fmElements.find((e) => e.tag === "00020010");
  if (tsElement === undefined || tsElement.vr !== "UI") {
    throw new DicomParseError(
      FATAL_CODES.INVALID_FILE_META,
      "Required File Meta element (0002,0010) Transfer Syntax UID is missing or not UI.",
      fmStart,
      buildSnippet(buffer, fmStart),
    );
  }

  const fileMeta: FileMeta = {
    transferSyntaxUID: trimUI(tsElement.value),
    ...projectOptional(fmElements, "00020002", "mediaStorageSOPClassUID"),
    ...projectOptional(fmElements, "00020003", "mediaStorageSOPInstanceUID"),
    ...projectOptionalRaw(fmElements, "00020001", "fileMetaInformationVersion"),
    ...projectOptional(fmElements, "00020012", "implementationClassUID"),
    ...projectOptional(fmElements, "00020013", "implementationVersionName"),
    ...projectOptional(fmElements, "00020016", "sourceApplicationEntityTitle"),
  };

  return { fileMeta, fileMetaEnd: cursor.position };
}

function readExplicitLeElement(cursor: ByteCursor): FmRawElement {
  const headerStart = cursor.position;
  const group = cursor.readUInt16();
  const element = cursor.readUInt16();
  const tag = (group.toString(16).padStart(4, "0") + element.toString(16).padStart(4, "0")).toUpperCase() as Tag;
  const vr = cursor.slice(2).toString("ascii") as VR;
  let length: number;
  let headerLength: number;
  if (LONG_FORM_VRS.has(vr)) {
    cursor.slice(2); // 2 reserved bytes — File Meta tolerated; reserved-byte non-zero check lives in dataset parsers.
    length = cursor.readUInt32();
    headerLength = 12;
  } else {
    length = cursor.readUInt16();
    headerLength = 8;
  }
  if (cursor.position + length > cursor.buffer.length) {
    throw new RangeError("File Meta element value extends past buffer end");
  }
  const value = cursor.slice(length);
  return { tag, vr, value, headerStart, headerLength };
}

function trimUI(buf: Buffer): string {
  let end = buf.length;
  while (end > 0 && (buf[end - 1] === 0x00 || buf[end - 1] === 0x20)) end--;
  return buf.subarray(0, end).toString("ascii");
}

function projectOptional(
  els: readonly FmRawElement[],
  tag: string,
  key: string,
): Partial<Record<string, string>> {
  const e = els.find((x) => x.tag === tag);
  if (e === undefined) return {};
  return { [key]: trimUI(e.value) };
}

function projectOptionalRaw(
  els: readonly FmRawElement[],
  tag: string,
  key: string,
): Partial<Record<string, Buffer>> {
  const e = els.find((x) => x.tag === tag);
  if (e === undefined) return {};
  return { [key]: e.value };
}

export function parseDicom(input: Buffer | Uint8Array | ArrayBuffer, options: ParseOptions = {}): Dataset {
  // First EMPTY_INPUT check — raw input length (D-13 dual-check, first half).
  if (rawInputIsEmpty(input)) {
    throw new DicomParseError(FATAL_CODES.EMPTY_INPUT, "Input is empty.", 0, "");
  }

  const buffer = normalizeInput(input);

  // Second EMPTY_INPUT check — after normalization (D-13 corner case for ArrayBuffer(0) views).
  if (buffer.length === 0) {
    throw new DicomParseError(FATAL_CODES.EMPTY_INPUT, "Input is empty after normalization.", 0, "");
  }

  // Build ParseContext. `copyValues: false` default per D-16.
  const warnings: DicomParseWarning[] = [];
  const ctx: ParseContext = {
    buffer,
    strict: options.strict === true,
    stripPreamble: options.stripPreamble ?? "tolerate",
    warnings,
    creators: new Map(),
    encodingContextStack: ["Root"],
    nestingDepth: 0,
    copyValues: options.copyValues === true,
    // Per D-02 + exactOptionalPropertyTypes: only set onWarning when defined.
    ...(options.onWarning !== undefined ? { onWarning: options.onWarning satisfies OnWarningCallback } : {}),
    // Per D-45 reservation — never set in Phase 2; field present in the type.
  };

  const strategy = TRANSFER_SYNTAX_PARSERS[tsUid];

      fileMetaEnd,
      tsName, // Per D-20 — human-readable name (when known) in snippet field.
    );
  }

}

function rawInputIsEmpty(input: Buffer | Uint8Array | ArrayBuffer): boolean {
  if (input instanceof ArrayBuffer) return input.byteLength === 0;
  return input.byteLength === 0;

function normalizeInput(input: Buffer | Uint8Array | ArrayBuffer): Buffer {
  if (Buffer.isBuffer(input)) return input;
  if (input instanceof Uint8Array) {
    return Buffer.from(input.buffer, input.byteOffset, input.byteLength);
  }
  // input is ArrayBuffer
  return Buffer.from(input);
}
