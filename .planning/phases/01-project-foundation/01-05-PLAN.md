**Step 3: Author `scripts/smoke.ts`** (driver). Uses `spawnSync` with array args — no shell:
```ts
#!/usr/bin/env tsx
/**
 * Phase 1 Plan 05 smoke driver — D-22.
 *
 * Validates the built dist/ artifacts by spawning two separate `node` processes:
 *   - one running test/smoke/esm/index.mjs (ESM consumer)
 *   - one running test/smoke/cjs/index.cjs (CJS consumer)
 *
 * Each harness imports VERSION + Dictionary from dist/, asserts known lookups,
 * and exits 0 on success. This driver returns 0 only if both succeed.
 *
 * Pre-requisites: `pnpm build` must have run; dist/index.mjs, dist/index.cjs,
 * dist/index.d.ts must exist. The driver verifies these and exits with a useful
 * message if not.
 */

import { spawnSync } from "node:child_process";
import { existsSync } from "node:fs";
import { join, dirname } from "node:path";
import { fileURLToPath } from "node:url";

const REPO_ROOT = join(dirname(fileURLToPath(import.meta.url)), "..");

const REQUIRED = [
  "dist/index.mjs",
  "dist/index.cjs",
  "dist/index.d.ts",
];

for (const rel of REQUIRED) {
  if (!existsSync(join(REPO_ROOT, rel))) {
    console.error(`[smoke] missing build artifact: ${rel}. Run 'pnpm build' first.`);
    process.exit(2);
  }
}

const harnesses: Array<{ label: string; path: string }> = [
  { label: "esm", path: "test/smoke/esm/index.mjs" },
  { label: "cjs", path: "test/smoke/cjs/index.cjs" },
];

let failed = 0;
for (const h of harnesses) {
  const r = spawnSync("node", [h.path], { cwd: REPO_ROOT, encoding: "utf8", shell: false });
  if (r.status !== 0) {
    console.error(`[smoke:${h.label}] FAIL exit=${r.status ?? "null"}`);
    if (r.stdout) console.error(r.stdout);
    if (r.stderr) console.error(r.stderr);
    failed++;
  } else {
    if (r.stdout) console.log(r.stdout.trim());
  }
}

if (failed > 0) {
  console.error(`[smoke] ${failed}/${harnesses.length} harness(es) failed.`);
  process.exit(1);
}

console.log("[smoke] OK — both ESM and CJS harnesses passed.");
```
